\paragraph{Admissible rules}
\begin{proofblock}
  \AXC{}
  \NOM{Ax}
  \UIC{$\seq[{ \link{x}{y} }]{ \tm[x]{A^\bot}, \tm[y]{A} }$}
\end{proofblock}
\begin{proofblock}
  \AXC{$\seq[{ P }]{ \Gamma, \tm[x]{\give[n]{A}} }$}
  \NOM{Exp}
  \UIC{$\seq[{ \expn{x}{x_1 \cdots x_n}P }]{ \Gamma, \tm[x_1]{A}\cdots\tm[x_n]{A} }$}
\end{proofblock}
\begin{proofblock}
  \AXC{$\seq[{ P }]{ \Gamma, \tm[x]{\take[n]{A}} }$}
  \AXC{$\seq[{ Q }]{ \Delta, \tm[y_1]{A^\bot}\cdots\tm[y_n]{A^\bot} }$}
  \NOM{Int}
  \BIC{$\seq[{ \intl{x}{y_1 \cdots y_n}(P \mid Q)}]{ \Gamma, \Delta } $}
\end{proofblock}
\begin{proofblock}
  \AXC{$\seq[{ P }]{ \Gamma, \tm[x]{A} }$}
  \AXC{$\seq[{ Q }]{ \Delta, \tm[x]{A^\bot} }$}
  \NOM{Cut}
  \BIC{$\seq[{ \cut{x}(P \mid Q) }]{ \Gamma, \Delta }$}
\end{proofblock}
