\begin{definition}[Negation]\label{def:nc-negation}
  We extend \cref{def:cp-negation} with the following cases:
  \[\!
    \begin{array}{lclclcl}
              \ty{(\take[n]{A})^\bot} &=& \ty{\give[n]{A^\bot}}
      &\quad& \ty{(\give[n]{A})^\bot} &=& \ty{\take[n]{A^\bot}}
    \end{array}
  \]
\end{definition}
