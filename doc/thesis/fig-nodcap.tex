      \SYM{\take[1]{}}

    \end{proofbox}

      \NOM{Contract}
      \UIC{$\seq[{ \subst{P}{x}{y} }]{ \Gamma , \tm[x]{\give[m+n]{A}} }$}
    \end{proofbox}
    \begin{proofbox}[1]
      \AXC{$\seq[{ P }]{ \Gamma , \tm[x]{\take[m]{A}} }$}
      \AXC{$\seq[{ Q }]{ \Delta , \tm[x]{\take[n]{A}} }$}
      \NOM{Pool}
      \BIC{$\seq[{ (P \mid Q) }]{ \Gamma , \Delta , \tm[x]{\take[m+n]{A}} }$}
    \end{proofbox}
