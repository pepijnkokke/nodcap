\begin{definition}[Types]\label{def:nc-types}
  We extend \cref{def:cp-types} with the following types:
  \[\!
    \begin{aligned}
      \ty{A}, \ty{B}, \ty{C}
           :=& \; \dots
      \\ \mid& \; \ty{\take[n]{A}} &&\text{pool of} \; n \; \text{clients}
      \\ \mid& \; \ty{\give[n]{A}} &&n \; \text{server interactions}
    \end{aligned}
  \]  
\end{definition}
