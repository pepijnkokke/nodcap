\begin{definition}[Evaluation contexts]\label{def:cp-evaluation-contexts}
  We define evaluation contexts as:
  \begin{align*}
    \tm{G}, \tm{H}
    := \tm{\Box}
    \mid \tm{\cpCut{x}{P}{G}}
    \mid \tm{\cpCut{x}{G}{P}}
  \end{align*}
\end{definition}
