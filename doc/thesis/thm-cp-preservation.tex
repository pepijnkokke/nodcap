\begin{theorem}[Preservation]\label{thm:cp-preservation}
  If \reducesto{P}{Q} and $\seq[{ P }]{ \Gamma }$, then $\seq[{ Q }]{ \Gamma }$.
\end{theorem}
\begin{proof}
  By induction on the structure of the reduction. See \cref{fig:cp-preservation}.
\end{proof}
