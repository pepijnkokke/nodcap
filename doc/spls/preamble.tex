% pi-calculus syntax

\newcommand{\new}[1]{{\color{tm}\nu #1.}}
\newcommand{\recv}[2]{{\color{tm}#1(#2).}}
\newcommand{\send}[3][]{{%
    \color{tm}
    \toks0={#1}%
    \edef\param{\the\toks0}%
    \edef\bound{b}%
    \edef\unbound{u}%
    \ifx\param\bound%
    \new{#3}\send[u]{#2}{#3}
    \else%
      \ifx\param\unbound%
      #2\langle #3 \rangle.
      \else%
      #2[#3].
      \fi
    \fi
  }}

\toks0={#1}%
\edef\param{\the\toks0}%

  \ensuremath{{\color{ty}#2}}%

  \ensuremath{{\color{tm}#1}\color{black}:{\color{ty}#2}}%
\fi}
