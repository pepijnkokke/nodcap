\begin{frame}
  \frametitle{Example: buying cake at the p\^{a}tisserie}

  \wide[4cm]{%
    \begin{center}
      \amb
      {\cut{x}({
          \quad \recv{x}{y}\john \quad \mid
          \quad \recv{x}{z}\mary \quad \mid
          \quad \send{x}{\cake}\send{x}{\nocake}\halt \quad
        })}
      {(\; \subst{\john}{\cake}{y}   \mid\: \subst{\mary}{\nocake}{z} \;)}
      {(\; \subst{\john}{\nocake}{y} \mid\: \subst{\mary}{\cake}{z}   \;)}
    \end{center}
  }
\end{frame}
